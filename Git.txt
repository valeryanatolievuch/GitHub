$ git branch - посмотреть все ветки

$ git branch название_ветки - создать новую ветку

$ git checkout -b название_ветки - создать новую ветку и перейти в неё

$ git checkout название_ветки - перейти в ветку

$ git push -u origin название_ветки - пушнуть ветку на внешний

$ cat >> название_файла_расширение - создать файл и отредактировать его

$ git merge название_ветки_с_инфой -m "сообщение" - вмержить инфу (Встаём в ту ветку куда вносим изменения)

$ git add "название_файла_расширение" - добавить содержимое файла в индекс
$ git add . - добавить всё содержимое в индекс
$ git commit -m "сообщение" - записать изменения в репозиторий
$ git push - запушить на внешний 

$ git fetch - загрузить объекты и ссылки из другого репозитория
